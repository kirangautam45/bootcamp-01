2. App.jsx - Main Component

This is the heart of your application that manages all state and API communication:

import { useState, useEffect } from 'react';
import NoteForm from './components/NoteForm';
import NoteList from './components/NoteList';
import './App.css';

const API_URL = 'http://localhost:5001/api/notes';

function App() {
  // STATE MANAGEMENT
  const [notes, setNotes] = useState([]);          // All notes array
  const [editingNote, setEditingNote] = useState(null); // Note being edited
  const [loading, setLoading] = useState(true);      // Loading state

  // Fetch notes when component mounts
  useEffect(() => {
    fetchNotes();
  }, []); // Empty array = run once on mount
  ...
}

React Hooks Explained:

Hook Purpose In This App
useState Manage component state Stores notes, editing state, loading state
useEffect Side effects (API calls, subscriptions) Fetches notes when app loads

API Functions in App.jsx:

fetchNotes() - GET all notes

Runs on component mount, fetches all notes from backend

const fetchNotes = async () => {
  try {
    const response = await fetch(API_URL);
    const data = await response.json();
    setNotes(data);  // Update state with fetched notes
  } catch (error) {
    console.error('Error fetching notes:', error);
  } finally {
    setLoading(false);  // Stop loading spinner
  }
};

addNote(note) - POST new note

Creates a new note and adds it to the beginning of the list

const addNote = async (note) => {
  try {
    const response = await fetch(API_URL, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(note)
    });
    const newNote = await response.json();
    setNotes([newNote, ...notes]); // Add to front of array
  } catch (error) {
    console.error('Error adding note:', error);
  }
};

updateNote(id, updatedNote) - PUT update

Updates an existing note by its ID

const updateNote = async (id, updatedNote) => {
  try {
    const response = await fetch(`${API_URL}/${id}`, {
      method: 'PUT',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(updatedNote)
    });
    const data = await response.json();
    // Replace old note with updated one in state
    setNotes(notes.map(note => note._id === id ? data : note));
    setEditingNote(null); // Clear editing mode
  } catch (error) {
    console.error('Error updating note:', error);
  }
};

deleteNote(id) - DELETE note

Removes a note from database and state

const deleteNote = async (id) => {
  try {
    await fetch(`${API_URL}/${id}`, { method: 'DELETE' });
    // Filter out deleted note from state
    setNotes(notes.filter(note => note._id !== id));
  } catch (error) {
    console.error('Error deleting note:', error);
  }
};

JSX Return (UI Structure):

return (
  <div className="app">

3. NoteForm.jsx - Create/Edit Form

Handles both creating new notes and editing existing ones:

import { useState } from 'react'
import './NoteForm.css'

// Available color options for notes
const COLORS = [
  '#ffffff',   // White
  '#fff9c4',   // Yellow
  '#c8e6c9',   // Green
  '#bbdefb',   // Blue
  '#d1c4e9',   // Purple
  '#ffccbc',   // Orange
]

function NoteForm({ onSubmit, editingNote, onCancel }) {
  // Initialize state with editing values or defaults
  const [title, setTitle] = useState(editingNote?.title || '')
  const [content, setContent] = useState(editingNote?.content || '')
  const [color, setColor] = useState(editingNote?.color || '#ffffff')

  const handleSubmit = (e) => {
    e.preventDefault()  // Prevent page reload
    if (!title.trim() || !content.trim()) return // Validation

    onSubmit({ title, content, color })

    // Clear form only if adding new note (not editing)
    if (!editingNote) {
      setTitle('')
      setContent('')
      setColor('#ffffff')
    }
  }
  ...
}

Key Concepts:

  • Optional Chaining (?.): editingNote?.title safely accesses properties even if editingNote is null
  • Controlled Components: Form inputs are controlled by React state (value + onChange)
  • Conditional Rendering: Cancel button only shows when editing

Color Picker Implementation:

<div className="color-picker">
  {COLORS.map((c) => (
    <button
      key={c}
      type="button"  // Prevents form submission
      className={`color-btn ${color === c ? 'selected' : ''}`}
      style={{ backgroundColor: c }}
      onClick={() => setColor(c)}
    />
  ))}
</div>

4. NoteList.jsx - Notes Grid

Displays all notes in a grid layout:

import NoteCard from './NoteCard';
import './NoteList.css';

function NoteList({ notes, onEdit, onDelete }) {
  // Show message if no notes exist
  if (notes.length === 0) {
    return (
      <div className="empty-state">
        <p>No notes yet. Create your first note!</p>
      </div>
    );
  }

  // Map through notes and render NoteCard for each
  return (
    <div className="note-list">
      {notes.map((note) => (
        <NoteCard
          key={note._id}  // Unique key for React's reconciliation
          note={note}
          onEdit={onEdit}
          onDelete={onDelete}
        />
      ))}
    </div>
  );
}

export default NoteList;

5. NoteCard.jsx - Individual Note

Displays a single note with edit/delete actions:

import { Pencil, Trash2 } from 'lucide-react';  // Icon imports
import './NoteCard.css';

function NoteCard({ note, onEdit, onDelete }) {
  // Helper function to format date
  const formatDate = (dateString) => {
    return new Date(dateString).toLocaleDateString('en-US', {
      month: 'short',
      day: 'numeric',
      year: 'numeric',
    });
  };

  return (
    <section
      className="note-card"
      style={{ backgroundColor: note.color || '#ffffff' }}
    >
      <h3 className="note-title">{note.title}</h3>
      <p className="note-content">{note.content}</p>

      <div className="note-footer">
        <span className="note-date">
          {formatDate(note.createdAt)}
        </span>

        <div className="note-actions">
          <button onClick={() => onEdit(note)} className="action-btn">
            <Pencil size={18} />
          </button>
          <button onClick={() => onDelete(note._id)} className="action-btn">
            <Trash2 size={18} />
          </button>
        </div>
      </div>
    </section>
  );
}

export default NoteCard;

Data Flow Diagram

1. User clicks "Add Note":

NoteForm → onSubmit() → App.addNote() → fetch POST → setNotes() → UI updates

2. User clicks Edit icon:

NoteCard → onEdit(note) → App.setEditingNote() → NoteForm receives editingNote prop → Form fills with data

3. User submits edit:

NoteForm → onSubmit() → App.updateNote() → fetch PUT → setNotes() → UI updates

4. User clicks Delete icon:

NoteCard → onDelete(id) → App.deleteNote() → fetch DELETE → setNotes(filter) → UI updates

Troubleshooting

CORS Error

If you see "Access-Control-Allow-Origin" error, make sure the backend is running with cors() middleware enabled.

Notes Not Loading

Check that:

  • Backend is running on port 5001
  • MongoDB is running
  • API_URL in App.jsx matches backend URL
Form Not Clearing

Make sure the key prop is set on NoteForm. This forces React to remount the component when switching between add/edit modes.
